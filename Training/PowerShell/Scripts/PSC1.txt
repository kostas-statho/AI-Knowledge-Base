<# ========================================================================
  PSC1.ps1
  Functions: Get-Users, Get-Roles, Get-UserInRole

  PURPOSE
    Read-only tooling for IAM-style lookups:
      • Users from dbo.[User]
      • Roles from dbo.Role
      • Memberships from dbo.UserInRole (joined with user & role info)

  AUTHENTICATION MODES (choose one explicitly)
      -Integrated                  → Windows Integrated Security
      -Credential                  → SWITCH that opens a GUI credential prompt
      -CredentialObject <PSCred>   → Use a prebuilt credential

   WHY A SWITCH FOR -Credential?
     PowerShell normally requires a value for [pscredential]. By using a switch
     and prompting internally, we get the ideal UX: write `-Credential` and
     receive a popup, with retry on wrong creds and graceful exit on cancel.

  FILTERING (Ask switches only)
      -AskUserID → prompt for a numeric UserID
      -AskRoleID → prompt for a numeric RoleID
    (No -UserID / -RoleID parameters exist to avoid binding errors.)

  LOGGING
    Per-run logs in:  C:\Users\OneIM\Desktop\PowerShell_Scripts\logs
    File name:        <FunctionName>_yyyyMMdd_HHmmss.log
    Levels:
      INFO  → Write-Host + file
      DEBUG → Write-Host + file
      TRACE → Write-Verbose + file   (visible only with -Verbose)
      ERROR → Write-Warning + file   (cleaner than Write-Error inside scriptblocks)

  PERFORMANCE/SAFETY NOTES
    - Reader ordinals cached to reduce field lookups
    - Connections/readers always closed in finally blocks
    - Parameterized SQL (no string concatenation for filters)
    - Encrypt=True; TrustServerCertificate=True to avoid TLS noise on dev

======================================================================== #>

Set-StrictMode -Version Latest

# ----------------------------- Common Helpers ---------------------------

function New-SqlConnectionString {
<#
.SYNOPSIS
  Build a SqlClient connection string for either Integrated or SQL authentication.
.DESCRIPTION
  Returns a properly configured connection string:
    - Integrated: uses Windows auth
    - SQL Auth : uses PSCredential (User/Password)
.PARAMETER Server
  SQL Server host\instance.
.PARAMETER Database
  Database name.
.PARAMETER Credential
  PSCredential containing SQL username/password (only for SQL auth).
.PARAMETER Integrated
  Switch for Windows Integrated Security.
.OUTPUTS
  [string] - A valid SqlClient connection string.
#>
[CmdletBinding()]
param(
    [Parameter(Mandatory)][ValidateNotNullOrEmpty()][string]$Server,
    [Parameter(Mandatory)][ValidateNotNullOrEmpty()][string]$Database,
    [pscredential]$Credential,
    [switch]$Integrated
)
    if ($Integrated) {
        return "Server=$Server;Database=$Database;Integrated Security=True;Encrypt=True;TrustServerCertificate=True;Connection Timeout=30;"
    }
    elseif ($Credential) {
        $u = $Credential.UserName
        $p = $Credential.GetNetworkCredential().Password
        return "Server=$Server;Database=$Database;User ID=$u;Password=$p;Encrypt=True;TrustServerCertificate=True;Connection Timeout=30;"
    }
    else {
        throw "Provide -Credential (SQL Auth) or -Integrated (Windows Auth)."
    }
}

function Test-SqlConnection {
<#
.SYNOPSIS
  Lightweight connection check that distinguishes wrong credentials.
.DESCRIPTION
  Attempts to open/close a SqlConnection. Returns $true on success.
  Returns $false on "Login failed…" (bad username/password).
  Rethrows other exceptions (DNS/network/availability).
#>
[CmdletBinding()]
param([Parameter(Mandatory)][string]$ConnectionString)
    try {
        $cn = [System.Data.SqlClient.SqlConnection]::new($ConnectionString)
        $cn.Open()
        $cn.Close()
        $cn.Dispose()
        return $true
    } catch {
        if ($_.Exception.Message -like "*Login failed*") { return $false }
        throw
    }
}

function New-Logger {
<#
.SYNOPSIS
  Create a per-run logger for the given function.
.DESCRIPTION
  - Writes to a timestamped file and mirrors to console.
  - TRACE only shows with -Verbose.
  - ERROR uses Write-Warning to avoid awkward scriptblock rendering in the console.
.OUTPUTS
  [scriptblock] - Closure accepting (Level, Message).
#>
[CmdletBinding()]
param([Parameter(Mandatory)][string]$FunctionName)
    $ts   = Get-Date -Format "yyyyMMdd_HHmmss"
    $dir  = "C:\Users\OneIM\Desktop\PowerShell_Scripts\logs"
    $path = Join-Path $dir ("{0}_{1}.log" -f $FunctionName,$ts)
    New-Item -ItemType Directory -Force -Path $dir -ErrorAction SilentlyContinue | Out-Null

    return {
        param([ValidateSet("INFO","DEBUG","TRACE","ERROR")][string]$Level,[string]$Message)
        $line = "{0} [{1}] {2}" -f (Get-Date -Format "yyyy-MM-dd HH:mm:ss"),$Level,$Message
        try { $line | Out-File -FilePath $path -Append -Encoding utf8 } catch { Write-Verbose "[LOGGER] $($_.Exception.Message)" }
        switch ($Level) {
            "INFO"  { Write-Host    $line }
            "DEBUG" { Write-Host    $line }
            "TRACE" { Write-Verbose $line }  # shown with -Verbose
            "ERROR" { Write-Warning $Message }
        }
    }.GetNewClosure()
}

function Get-ConnectionStringWithRetry {
<#
.SYNOPSIS
  Centralized auth logic with retry & cancel handling for SQL Auth; pass-through for Integrated.
.DESCRIPTION
  - Integrated: immediately returns a connection string.
  - SQL Auth:
      * If allowed to prompt and no PSCredential yet → show GUI prompt.
      * If user CANCELS prompt → log & return $null.
      * If credentials are WRONG → warn and re-prompt until success or cancel.
.PARAMETER AllowPromptWhenMissing
  $true  → show credential prompt if we lack a credential
  $false → if no credential is present, exit gracefully (no extra prompts)
#>
[CmdletBinding()]
param(
    [Parameter(Mandatory)][string]$SqlServer,
    [Parameter(Mandatory)][string]$SqlDatabase,
    [pscredential]$Credential,
    [switch]$Integrated,
    [Parameter(Mandatory)][scriptblock]$Log,
    [Parameter(Mandatory)][string]$FunctionName,
    [bool]$AllowPromptWhenMissing = $true
)
    if ($Integrated) {
        return (New-SqlConnectionString -Server $SqlServer -Database $SqlDatabase -Integrated)
    }

    $credLocal = $Credential
    $attempted = $false

    while ($true) {
        if (-not $credLocal) {
            if (-not $AllowPromptWhenMissing) {
                & $Log 'INFO' "$($FunctionName): Authentication canceled by user."
                return $null
            }
            $credLocal = Get-Credential -Message "Enter SQL Server credentials for $SqlServer"
            if (-not $credLocal) {
                & $Log 'INFO' "$($FunctionName): Authentication canceled by user."
                return $null
            }
        }

        $connStr = New-SqlConnectionString -Server $SqlServer -Database $SqlDatabase -Credential $credLocal
        $ok = $false
        try { $ok = Test-SqlConnection -ConnectionString $connStr } catch { throw }
        $attempted = $true

        if ($ok) { return $connStr }

        Write-Warning "Wrong credentials. Please try again."
        $credLocal = $null     # force re-prompt
        $AllowPromptWhenMissing = $true
    }
}

function Read-IntInteractive {
<#
.SYNOPSIS
  Prompt until a valid integer is entered.
.PARAMETER Name
  Friendly label to show in the prompt.
#>
[CmdletBinding()]
param([Parameter(Mandatory)][string]$Name)
    while ($true) {
        $inputText = Read-Host "Enter $Name (numeric)"
        $parsed = 0
        if ([int]::TryParse($inputText, [ref]$parsed)) { return $parsed }
        Write-Warning "Please enter a valid number for $Name."
    }
}

function Format-CriteriaMessage {
<#
.SYNOPSIS
  Build a compact criteria description including only applied filters.
.DESCRIPTION
  Returns "{}" if no filters are applied; otherwise "{ key=value, ... }".
#>
[CmdletBinding()]
param(
    [Nullable[int]]$UserID,
    [Nullable[int]]$RoleID
)
    $parts = @()
    if ($null -ne $UserID) { $parts += "UserID=$UserID" }
    if ($null -ne $RoleID) { $parts += "RoleID=$RoleID" }
    if ($parts.Count -eq 0) { return "{}" }
    return "{ " + ($parts -join ", ") + " }"
}

# ------------------------------ Get-Users --------------------------------

function global:Get-Users {
<#
.SYNOPSIS
  Retrieve users from dbo.[User].
.DESCRIPTION
  Returns objects with: UserID, FirstName, LastName, EmailAddress, JobDescription.
  Use -AskUserID to filter interactively by numeric user key; without it, returns all users.
.PARAMETER SqlServer
  SQL Server host\instance.
.PARAMETER SqlDatabase
  Database name.
.PARAMETER Credential
  SWITCH → open a credential GUI prompt (SQL Auth).
.PARAMETER CredentialObject
  A PSCredential you already created (SQL Auth).
.PARAMETER Integrated
  Use Windows Integrated Security.
.PARAMETER AskUserID
  Prompt to input a numeric UserID filter.
.EXAMPLE
  Get-Users -SqlServer "StathopoulosK" -SqlDatabase "CCC_PSC" -Integrated
.EXAMPLE
  Get-Users -SqlServer "StathopoulosK" -SqlDatabase "CCC_PSC" -Credential -AskUserID   # prompts for UserID
#>
[CmdletBinding()]
param(
    [Parameter(Mandatory)][string]$SqlServer,
    [Parameter(Mandatory)][string]$SqlDatabase,
    [Alias('Credential')][switch]$PromptCredential,
    [pscredential]$CredentialObject,
    [switch]$Integrated,
    [switch]$AskUserID,
    [string]$ObjectClass = 'dbo.[User]'
)
    $Log = New-Logger -FunctionName 'Get-Users'
    & $Log 'INFO' "START Get-Users (Object: $ObjectClass)"
    try {
        # explicit auth choice required
        if (-not $Integrated -and -not $PromptCredential -and -not $PSBoundParameters.ContainsKey('CredentialObject')) {
            & $Log 'ERROR' "No authentication specified. Use -Integrated, -Credential (prompt), or -CredentialObject <PSCredential>."
            return
        }

        # (optional) prompt for UserID if requested
        [Nullable[int]]$UserID = $null
        if ($AskUserID) { $UserID = Read-IntInteractive -Name 'UserID' }

        # decide whether to prompt for SQL credentials when missing
        $allowPromptWhenMissing = $PromptCredential -or (-not $PSBoundParameters.ContainsKey('CredentialObject'))

        $connStr = Get-ConnectionStringWithRetry -SqlServer $SqlServer -SqlDatabase $SqlDatabase `
                     -Credential $CredentialObject -Integrated:$Integrated -Log $Log -FunctionName 'Get-Users' `
                     -AllowPromptWhenMissing:$allowPromptWhenMissing
        if (-not $connStr) { return } # user canceled

        # build query (parameterized)
        $filtering = $AskUserID
        $q = if ($filtering) {
@"
SELECT UserID, FirstName, LastName, EmailAddress, JobDescription
FROM $ObjectClass WHERE UserID=@UserID
ORDER BY UserID;
"@
        } else {
@"
SELECT UserID, FirstName, LastName, EmailAddress, JobDescription
FROM $ObjectClass
ORDER BY UserID;
"@
        }
        & $Log 'TRACE' "Query: $q"

        Add-Type -AssemblyName System.Data -ErrorAction SilentlyContinue
        $cn = [System.Data.SqlClient.SqlConnection]::new($connStr)
        $reader = $null
        try {
            $cn.Open()
            $cmd = $cn.CreateCommand()
            $cmd.CommandText = $q
            if ($filtering) {
                $null = $cmd.Parameters.Add("@UserID",[System.Data.SqlDbType]::Int)
                $cmd.Parameters["@UserID"].Value = $UserID
            }

            $rows = [System.Collections.Generic.List[object]]::new()
            & $Log 'DEBUG' "Query running..."
            $reader = $cmd.ExecuteReader()

            # cache ordinals once (micro-optimization)
            $oUID = $reader.GetOrdinal("UserID")
            $oFN  = $reader.GetOrdinal("FirstName")
            $oLN  = $reader.GetOrdinal("LastName")
            $oEM  = $reader.GetOrdinal("EmailAddress")
            $oJD  = $reader.GetOrdinal("JobDescription")

            while ($reader.Read()) {
                $rows.Add([pscustomobject]@{
                    UserID        = $reader.GetValue($oUID)
                    FirstName     = $reader.GetValue($oFN)
                    LastName      = $reader.GetValue($oLN)
                    EmailAddress  = $reader.GetValue($oEM)
                    JobDescription= $reader.GetValue($oJD)
                })
            }
        } finally {
            if ($reader) { $reader.Close() }
            if ($cn.State -eq 'Open') { $cn.Close() }
            $cn.Dispose()
        }

        if ($rows.Count -eq 0) {
            $crit = Format-CriteriaMessage -UserID:($(if ($filtering) { $UserID } else { $null })) -RoleID:$null
            & $Log 'ERROR' "No rows found with criteria $crit"
            return @()
        }

    } catch { & $Log 'ERROR' ("Get-Users failed: {0}" -f $_.Exception.Message); throw }
    & $Log 'INFO' "END Get-Users"
    return $rows
}

function global:Get-Roles {
<#
.SYNOPSIS
  Retrieve roles from dbo.Role.
.DESCRIPTION
  Returns objects with: RoleID, RoleName. Use -AskRoleID to filter; otherwise returns all.
.PARAMETER AskRoleID
  Prompt to input a numeric RoleID filter.
#>
[CmdletBinding()]
param(
    [Parameter(Mandatory)][string]$SqlServer,
    [Parameter(Mandatory)][string]$SqlDatabase,
    [Alias('Credential')][switch]$PromptCredential,
    [pscredential]$CredentialObject,
    [switch]$Integrated,
    [switch]$AskRoleID,
    [string]$ObjectClass = 'dbo.Role'
)
    $Log = New-Logger -FunctionName 'Get-Roles'
    & $Log 'INFO' "START Get-Roles (Object: $ObjectClass)"
    try {
        if (-not $Integrated -and -not $PromptCredential -and -not $PSBoundParameters.ContainsKey('CredentialObject')) {
            & $Log 'ERROR' "No authentication specified. Use -Integrated, -Credential (prompt), or -CredentialObject <PSCredential>."
            return
        }

        [Nullable[int]]$RoleID = $null
        if ($AskRoleID) { $RoleID = Read-IntInteractive -Name 'RoleID' }

        $allowPromptWhenMissing = $PromptCredential -or (-not $PSBoundParameters.ContainsKey('CredentialObject'))

        $connStr = Get-ConnectionStringWithRetry -SqlServer $SqlServer -SqlDatabase $SqlDatabase `
                     -Credential $CredentialObject -Integrated:$Integrated -Log $Log -FunctionName 'Get-Roles' `
                     -AllowPromptWhenMissing:$allowPromptWhenMissing
        if (-not $connStr) { return }

        $filtering = $AskRoleID
        $q = if ($filtering) {
            "SELECT RoleID, RoleName FROM $ObjectClass WHERE RoleID=@RoleID"
        } else {
            "SELECT RoleID, RoleName FROM $ObjectClass ORDER BY RoleID"
        }
        & $Log 'TRACE' "Query: $q"

        Add-Type -AssemblyName System.Data -ErrorAction SilentlyContinue
        $cn = [System.Data.SqlClient.SqlConnection]::new($connStr)
        $reader = $null
        try {
            $cn.Open()
            $cmd = $cn.CreateCommand()
            $cmd.CommandText = $q
            if ($filtering) {
                $null = $cmd.Parameters.Add("@RoleID",[System.Data.SqlDbType]::Int)
                $cmd.Parameters["@RoleID"].Value = $RoleID
            }

            $rows = [System.Collections.Generic.List[object]]::new()
            & $Log 'DEBUG' "Query running..."
            $reader = $cmd.ExecuteReader()

            $oRID = $reader.GetOrdinal("RoleID")
            $oRNM = $reader.GetOrdinal("RoleName")
            while ($reader.Read()) {
                $rows.Add([pscustomobject]@{
                    RoleID   = $reader.GetValue($oRID)
                    RoleName = $reader.GetValue($oRNM)
                })
            }
        } finally {
            if ($reader) { $reader.Close() }
            if ($cn.State -eq 'Open') { $cn.Close() }
            $cn.Dispose()
        }

        if ($rows.Count -eq 0) {
            $crit = Format-CriteriaMessage -UserID:$null -RoleID:($(if ($filtering) { $RoleID } else { $null }))
            & $Log 'ERROR' "No rows found with criteria $crit"
            return @()
        }

    } catch { & $Log 'ERROR' ("Get-Roles failed: {0}" -f $_.Exception.Message); throw }
    & $Log 'INFO' "END Get-Roles"
    return $rows
}

# --------------------------- Get-UserInRole ------------------------------

function global:Get-UserInRole {
<#
.SYNOPSIS
  Retrieve user-role memberships with user & role details.
.DESCRIPTION
  Joins dbo.UserInRole with dbo.[User] and dbo.Role.
  Use -AskUserID and/or -AskRoleID to add filters; no switches returns all.
#>
[CmdletBinding()]
param(
    [Parameter(Mandatory)][string]$SqlServer,
    [Parameter(Mandatory)][string]$SqlDatabase,
    [Alias('Credential')][switch]$PromptCredential,
    [pscredential]$CredentialObject,
    [switch]$Integrated,
    [switch]$AskUserID,
    [switch]$AskRoleID,
    [string]$MembershipObject = 'dbo.UserInRole',
    [string]$UsersObject      = 'dbo.[User]',
    [string]$RolesObject      = 'dbo.Role'
)
    $Log = New-Logger -FunctionName 'Get-UserInRole'
    & $Log 'INFO' "START Get-UserInRole"
    try {
        if (-not $Integrated -and -not $PromptCredential -and -not $PSBoundParameters.ContainsKey('CredentialObject')) {
            & $Log 'ERROR' "No authentication specified. Use -Integrated, -Credential (prompt), or -CredentialObject <PSCredential>."
            return
        }

        [Nullable[int]]$UserID = $null
        [Nullable[int]]$RoleID = $null
        if ($AskUserID) { $UserID = Read-IntInteractive -Name 'UserID' }
        if ($AskRoleID) { $RoleID = Read-IntInteractive -Name 'RoleID' }

        $allowPromptWhenMissing = $PromptCredential -or (-not $PSBoundParameters.ContainsKey('CredentialObject'))

        $connStr = Get-ConnectionStringWithRetry -SqlServer $SqlServer -SqlDatabase $SqlDatabase `
                     -Credential $CredentialObject -Integrated:$Integrated -Log $Log -FunctionName 'Get-UserInRole' `
                     -AllowPromptWhenMissing:$allowPromptWhenMissing
        if (-not $connStr) { return }

        $where = @()
        if ($AskUserID) { $where += "uir.UserID=@UserID" }
        if ($AskRoleID) { $where += "uir.RoleID=@RoleID" }
        $whereSql = if ($where.Count) { "WHERE " + ($where -join " AND ") } else { "" }

        $q = @"
SELECT
    uir.UserID, u.FirstName, u.LastName, u.EmailAddress, u.JobDescription,
    uir.RoleID, r.RoleName
FROM $MembershipObject AS uir
LEFT JOIN $UsersObject AS u ON u.UserID = uir.UserID
LEFT JOIN $RolesObject AS r ON r.RoleID = uir.RoleID
$whereSql
ORDER BY uir.UserID, uir.RoleID;
"@
        & $Log 'TRACE' "Query: $q"

        Add-Type -AssemblyName System.Data -ErrorAction SilentlyContinue
        $cn = [System.Data.SqlClient.SqlConnection]::new($connStr)
        $reader = $null
        try {
            $cn.Open()
            $cmd = $cn.CreateCommand()
            $cmd.CommandText = $q
            if ($AskUserID) { $null=$cmd.Parameters.Add("@UserID",[System.Data.SqlDbType]::Int); $cmd.Parameters["@UserID"].Value=$UserID }
            if ($AskRoleID) { $null=$cmd.Parameters.Add("@RoleID",[System.Data.SqlDbType]::Int); $cmd.Parameters["@RoleID"].Value=$RoleID }

            $rows = [System.Collections.Generic.List[object]]::new()
            & $Log 'DEBUG' "Query running..."
            $reader = $cmd.ExecuteReader()

            $oUID=$reader.GetOrdinal("UserID");$oFN=$reader.GetOrdinal("FirstName");$oLN=$reader.GetOrdinal("LastName")
            $oEM=$reader.GetOrdinal("EmailAddress");$oJD=$reader.GetOrdinal("JobDescription")
            $oRID=$reader.GetOrdinal("RoleID");$oRNM=$reader.GetOrdinal("RoleName")

            while ($reader.Read()){
                $rows.Add([pscustomobject]@{
                    UserID        = $reader.GetValue($oUID)
                    FirstName     = $reader.GetValue($oFN)
                    LastName      = $reader.GetValue($oLN)
                    EmailAddress  = $reader.GetValue($oEM)
                    JobDescription= $reader.GetValue($oJD)
                    RoleID        = $reader.GetValue($oRID)
                    RoleName      = $reader.GetValue($oRNM)
                })
            }
        } finally {
            if ($reader) { $reader.Close() }
            if ($cn.State -eq 'Open') { $cn.Close() }
            $cn.Dispose()
        }

        if ($rows.Count -eq 0) {
            $crit = Format-CriteriaMessage -UserID:($(if ($AskUserID) { $UserID } else { $null })) `
                                           -RoleID:($(if ($AskRoleID) { $RoleID } else { $null }))
            & $Log 'ERROR' "No rows found with criteria $crit"
            return @()
        }

    } catch { & $Log 'ERROR' ("Get-UserInRole failed: {0}" -f $_.Exception.Message); throw }
    & $Log 'INFO' "END Get-UserInRole"
    return $rows
}
