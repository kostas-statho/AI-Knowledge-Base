      ]]></CustomCommand>

      <CustomCommand Name="Get-Authorization"><![CDATA[
        param([string]$SqlServer,[string]$SqlDatabase,[string]$username,[string]$password)
        $global:SqlServer   = $SqlServer
        $global:SqlDatabase = $SqlDatabase
        $global:SqlUser     = $username
        $global:SqlPass     = $password
      ]]></CustomCommand>
